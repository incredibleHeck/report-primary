/**
 * Hydrates the central library cache with local container properties.
 * Always fires prior to cross-project execution runs.
 */
function _hydrateClientScriptProperties() {
  if (typeof HecTechVault !== 'undefined' && HecTechVault.hydrateClientScriptProperties) {
    HecTechVault.hydrateClientScriptProperties(PropertiesService.getScriptProperties().getProperties());
  }
}

 * 🟢 REQUIRED PERMISSIONS MAPPING (Maintains active compilation scopes)
 */

  if (false) {

/**
 * Synchronizes library-bundled defaults safely into local storage parameters
 * without altering active configuration keys or secret credentials.
 */
function _applyBundledVaultDefaultsIfMissing() {
  if (typeof HecTechVault === 'undefined' || !HecTechVault.getBundledContainerPropertyDefaults) return;
  
  var patch = HecTechVault.getBundledContainerPropertyDefaults();
  if (!patch || typeof patch !== 'object') return;
  
  var props = PropertiesService.getScriptProperties();
  var cur = props.getProperties();
  var writeBuffer = {};
  var n = 0;
  
  Object.keys(patch).forEach(function (k) {
    var v = patch[k];
    if (v == null || v === '') return;
    var existing = cur[k];
    if (existing != null && String(existing).trim() !== '') return;
    writeBuffer[k] = String(v);
    n++;
  });
  
  if (n > 0) {
    props.setProperties(writeBuffer); // Optimized atomic batch write operation
    var ver = HecTechVault.getBundledContainerDefaultsVersion ? HecTechVault.getBundledContainerDefaultsVersion() : '?';
    console.log('HecTech Master: Populated ' + n + ' baseline default configurations [Bundle v' + ver + '].');
  }
}

function onOpen() {
  if (typeof HecTechVault !== 'undefined') {
    _applyBundledVaultDefaultsIfMissing();
    _hydrateClientScriptProperties();
    HecTechVault.onOpen();
  } else {
    SpreadsheetApp.getUi().createMenu('HecTech AI 🎓 (Error)')
      .addItem('⚠️ Library Link Broken', 'showLibraryError')
      .addToUi();
  }
}

function showLibraryError() {
  SpreadsheetApp.getUi().alert(
    "Critical System Link Disconnected",
    "Please attach the master engine vault library under project configurations using the target identifier: HecTechVault",
    SpreadsheetApp.getUi().ButtonSet.OK
  );
}
