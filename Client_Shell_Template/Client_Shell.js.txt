// =========================================================================
// HECTECH CLIENT SHELL (this vault project only)
// =========================================================================
// Bridge between the Google Sheet and this repo’s vault library.
// Add exactly one library; identifier must be HecTechVault.
//
// Class settings, subject context (CTX_*), and other per-user keys live in
// THIS project’s Script Properties — not the library’s. The library is
// hydrated from these properties before each menu action.
//
// INSTRUCTIONS:
// 1. Libraries (+) → paste this vault’s Script ID → latest version.
// 2. Set Identifier to exactly: HecTechVault
//
// Bundled defaults: the library can ship non-secret default Script Property keys. On open,
// missing/blank keys are filled from the library version (secrets still set per school).
// =========================================================================

function _hydrateClientScriptProperties() {
  if (typeof HecTechVault !== 'undefined' && HecTechVault.hydrateClientScriptProperties) {
    HecTechVault.hydrateClientScriptProperties(PropertiesService.getScriptProperties().getProperties());
  }
}

/**
 * 🟢 REQUIRED PERMISSIONS (DO NOT DELETE)
 */
function _forcePermissions() {
  if (false) {
    DriveApp.getFiles();
    MailApp.getRemainingDailyQuota();
    UrlFetchApp.fetch("https://google.com");
    ScriptApp.getOAuthToken();
    CacheService.getUserCache();
    PropertiesService.getScriptProperties();
  }
}

/**
 * Fills container Script Properties from library-bundled defaults only where a key is
 * missing or empty. Does not overwrite tokens or any non-empty value.
 */
function _applyBundledVaultDefaultsIfMissing() {
  if (typeof HecTechVault === 'undefined' || !HecTechVault.getBundledContainerPropertyDefaults) return;
  var patch = HecTechVault.getBundledContainerPropertyDefaults();
  if (!patch || typeof patch !== 'object') return;
  var props = PropertiesService.getScriptProperties();
  var cur = props.getProperties();
  var n = 0;
  Object.keys(patch).forEach(function (k) {
    var v = patch[k];
    if (v == null || v === '') return;
    var existing = cur[k];
    if (existing != null && String(existing).trim() !== '') return;
    props.setProperty(k, String(v));
    n++;
  });
  if (n > 0) {
    var ver = HecTechVault.getBundledContainerDefaultsVersion
      ? HecTechVault.getBundledContainerDefaultsVersion()
      : '?';
    console.log('HecTech: merged ' + n + ' bundled script property default(s) (defaults bundle v' + ver + ').');
  }
}

function onOpen() {
  if (typeof HecTechVault !== 'undefined') {
    _applyBundledVaultDefaultsIfMissing();
    _hydrateClientScriptProperties();
    HecTechVault.onOpen();
  } else {
    SpreadsheetApp.getUi().createMenu('HecTech AI 🎓 (Error)')
      .addItem('⚠️ Library Not Connected', 'showLibraryError')
      .addToUi();
  }
}

function showLibraryError() {
  SpreadsheetApp.getUi().alert(
    "Library Connection Error",
    "Add this vault under Libraries with identifier HecTechVault (exactly one library).",
    SpreadsheetApp.getUi().ButtonSet.OK
  );
}

function getSettingsData() {
  var map = PropertiesService.getScriptProperties().getProperties();
  var ssId = SpreadsheetApp.getActiveSpreadsheet().getId();
  return HecTechVault.getSettingsFromPropertyMap(map, ssId);
}

function saveSettingsData(settings) {
  var ssId = SpreadsheetApp.getActiveSpreadsheet().getId();
  var payload = HecTechVault.buildSettingsSaveProperties(settings, ssId);
  PropertiesService.getScriptProperties().setProperties(payload);
  return HecTechVault.afterClientSettingsSaved(settings);
}

function getSubjectContext() {
  var map = PropertiesService.getScriptProperties().getProperties();
  return HecTechVault.getSubjectContextFromPropertyMap(map);
}

function saveSubjectContext(data) {
  var entry = HecTechVault.buildSubjectContextSaveEntry(data);
  PropertiesService.getScriptProperties().setProperty(entry.key, entry.value);
  return entry.message;
}
